import PyPDF4

# Open the input PDF file
input_file = open('example.pdf', 'rb')

# Create a PDF reader object
pdf_reader = PyPDF4.PdfFileReader(input_file)

# Get the root outline object (i.e. the table of contents)
root_outline = pdf_reader.outlines

# Define a function to recursively print the outlines
def print_outlines(outline, level=0):
    for child in outline:
        title = child.title
        if title is not None:
            print(f"{' ' * (level * 4)}{title}")
            print_outlines(child, level=level+1)

# Print the table of contents
print_outlines(root_outline)

# Close the input file
input_file.close()
